effect
{
    Name: "Testing the AddPossibilities",
    Params:{
        Auxilio: Bool,
        Numerin: Number,
        Letra: String,
        Letraa: String
    },
	Action: (galletica, context) => 
    {
        inicio=Numerin;
        for target in galletica.Find((unit)=> (unit.Power>3 && (Auxilio|| Letra== Letraa)))
        {
            target.Power= inicio++;
        }
    }

}



card {
    Type: "Plata",
    Name: "Beluga",
    Faction: "Prueba",
    Power: 5,
    Range: ["Melee", "Range"],
    OnActivation: 
    [
        {
            Effect:
            {
               	Name: "Testing the AddPossibilities",
                Auxilio: false|| true,
                Numerin: 2^2^3,
                Letra: "Ingenieros Celestiales",
                Letraa: "Compilada"
            },
            Selector:
            {
                Source: "otherhand",
                Single: false,
                Predicate: (unit) => true
            },
        }
    ]
}
